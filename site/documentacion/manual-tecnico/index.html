    1.3 Patrones Implementados¶

    1️⃣ Blueprint Pattern (Modularización)¶

    • Cada funcionalidad en un Blueprint independiente
    • Permite escalabilidad y desarrollo en paralelo
    • Routes aislados por dominio

    2️⃣ MVC (Model-View-Controller)¶

    • Models: models.py (SQLAlchemy)
    • Views: templates/ (Jinja2)
    • Controllers: *_routes.py (Blueprints)

    3️⃣ Proxy Pattern (Conexiones DB)¶

    • Reconexión automática a PostgreSQL
    • Manejo de errores de conexión transparente

    4️⃣ Context Processors (Datos globales)¶

    • Inyectan datos en todos los templates
    • Ejemplo: notificaciones sin leer, usuario actual

    5️⃣ Helper Layer (Reutilización código)¶

    • Funciones compartidas entre blueprints
    • Ejemplo: crear_notificacion() desde múltiples módulos

    3. BASE DE DATOS – MODELO ENTIDAD-RELACIÓN¶

    3.1 Tablas Principales¶

    👥 usuarios¶

    CREATE TABLE usuarios (
  id SERIAL PRIMARY KEY,
  nombre VARCHAR(100) NOT NULL,
  apellidos VARCHAR(100),
  telefono VARCHAR(20),
  username VARCHAR(50) UNIQUE NOT NULL,
  email VARCHAR(100) UNIQUE NOT NULL,
  password VARCHAR(255),
  foto VARCHAR(255),
  fecha_registro TIMESTAMP DEFAULT NOW(),
  rol VARCHAR(20) DEFAULT 'comprador' 
      CHECK (rol IN ('comprador','vendedor','admin','superadmin')),
  estado VARCHAR(20) DEFAULT 'activo' 
      CHECK (estado IN ('activo','inactivo','bloqueado')),
  ultima_sesion TIMESTAMP,
  nombre_empresa VARCHAR(200)
);

    🚗 vehiculos¶

    CREATE TABLE vehiculos (
  id SERIAL PRIMARY KEY,
  id_usuario INT REFERENCES usuarios(id) ON DELETE SET NULL,
  usuario_id INT REFERENCES usuarios(id) ON DELETE SET NULL, -- alias heredado
  id_tipo INT REFERENCES tipos_vehiculos(id),
  id_marca INT REFERENCES marcas(id),
  id_modelo INT REFERENCES modelos(id),
  version VARCHAR(100),
  id_color INT REFERENCES colores(id),
  anio INT,
  placa VARCHAR(10),
  ciudad_placa VARCHAR(100),
  ciudad_venta VARCHAR(100),
  dueno VARCHAR(10) DEFAULT 'Si',          -- 'Si' | 'No'
  kilometraje INT,
  transmision VARCHAR(50) DEFAULT 'Manual', -- 'Manual' | 'Automatico'
  combustible VARCHAR(50) DEFAULT 'Gasolina',
  motor VARCHAR(50),
  traccion VARCHAR(50),
  puertas INT,
  precio NUMERIC(12,2),
  negociable VARCHAR(10) DEFAULT 'Si',
  imagen VARCHAR(255),                      -- imagen principal (legacy)
  slug VARCHAR(255) UNIQUE,
  descripcion TEXT,
  fecha_publicacion TIMESTAMP DEFAULT NOW(),
  estado VARCHAR(20) DEFAULT 'disponible',  -- 'activo' | 'disponible' | 'vendido' | 'pausado'
  vistas INT DEFAULT 0,
  -- Monetización: anuncios destacados
  plan_destacado BOOLEAN DEFAULT FALSE,
  comprobante_pago VARCHAR(255),
  estado_pago VARCHAR(20) DEFAULT 'ninguno', -- 'ninguno'|'pendiente'|'aprobado'|'rechazado'|'expirado'
  fecha_fin_destacado TIMESTAMP
);

-- ⚠️ Nota: el sistema filtra por estado = 'activo' en búsquedas públicas
--  y por estado IN ('disponible', 'activo') en el módulo IA.
--  Se recomienda unificar a un único valor en una próxima migración.

    🏪 perfil_vendedor¶

    CREATE TABLE perfil_vendedor (
  id SERIAL PRIMARY KEY,
  usuario_id INT UNIQUE REFERENCES usuarios(id) ON DELETE CASCADE,
  -- Información comercial
  nombre_tienda VARCHAR(100) NOT NULL,
  email_comercial VARCHAR(100) NOT NULL,
  telefono_contacto VARCHAR(20) NOT NULL,
  descripcion TEXT,
  -- Información legal (KYC)
  numero_id VARCHAR(50) NOT NULL,          -- Cédula/NIT
  -- Documentos de verificación y branding
  foto_id_frontal VARCHAR(255),            -- Imagen frontal del documento
  foto_id_trasera VARCHAR(255),            -- Imagen trasera del documento
  foto_portada VARCHAR(255),               -- Banner público de la tienda
  -- Estado y verificación
  estado_verificacion VARCHAR(20) DEFAULT 'pendiente'
      CHECK (estado_verificacion IN ('pendiente','aprobado','rechazado')),
  -- Datos bancarios (opcionales)
  banco VARCHAR(50),
  numero_cuenta VARCHAR(50)
);

    💬 conversaciones + mensajes¶

    CREATE TABLE conversaciones (
  id SERIAL PRIMARY KEY,
  id_vehiculo INT REFERENCES vehiculos(id) ON DELETE CASCADE,
  id_comprador INT REFERENCES usuarios(id) ON DELETE CASCADE,
  id_vendedor INT REFERENCES usuarios(id) ON DELETE CASCADE,
  fecha_creacion TIMESTAMP DEFAULT NOW(),
  ultima_actividad TIMESTAMP DEFAULT NOW(),

  UNIQUE (id_vehiculo, id_comprador, id_vendedor),
  INDEX idx_comprador (id_comprador),
  INDEX idx_vendedor (id_vendedor)
);

CREATE TABLE mensajes (
  id SERIAL PRIMARY KEY,
  id_conversacion INT REFERENCES conversaciones(id) ON DELETE CASCADE,
  id_remitente INT REFERENCES usuarios(id) ON DELETE CASCADE,
  contenido TEXT NOT NULL,
  fecha_envio TIMESTAMP DEFAULT NOW(),
  leido BOOLEAN DEFAULT FALSE,

  INDEX idx_conversacion (id_conversacion),
  INDEX idx_remitente (id_remitente)
);

    🔔 notificaciones¶

    CREATE TABLE notificaciones (
  id SERIAL PRIMARY KEY,
  id_usuario INT REFERENCES usuarios(id) ON DELETE CASCADE,
  tipo VARCHAR(50) NOT NULL,
  titulo VARCHAR(100) NOT NULL,
  mensaje TEXT NOT NULL,
  url_accion VARCHAR(255),
  id_relacion INT,
  leida BOOLEAN DEFAULT FALSE,
  fecha_creacion TIMESTAMP DEFAULT NOW(),
  fecha_leida TIMESTAMP,

  INDEX idx_usuario_leida (id_usuario, leida)
);

    📊 logs_admin¶

    CREATE TABLE logs_admin (
  id SERIAL PRIMARY KEY,
  id_admin INT,
  nombre_admin VARCHAR(200),
  accion VARCHAR(100) NOT NULL,
  entidad VARCHAR(50),
  id_entidad INT,
  descripcion TEXT,
  ip VARCHAR(50),
  fecha TIMESTAMP DEFAULT NOW(),

  INDEX idx_admin_fecha (id_admin, fecha)

    @app.route('/api/vehiculos')
def listar_vehiculos():

    # Usar offset + limit eficientemente
    total = db.session.query(Vehiculo).count()
    vehiculos = db.session.query(Vehiculo)\
        .offset((pagina - 1) * por_pagina)\
        .limit(por_pagina)\

        'total_paginas': ceil(total / por_pagina)
    })

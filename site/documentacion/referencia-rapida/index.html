    🎯 INICIO RÁPIDO¶

    Setup Local (5 minutos)¶

    # 1. Clonar
git clone https://github.com/drivemarket/drivemarket.git && cd drivemarket

# 2. Entorno
python -m venv .venv && source .venv/bin/activate
pip install -r requirements.txt

# 3. Config
cp .env.example .env  # Editar credenciales

# 4. BD
createdb todoen1unos
psql todoen1unos < todoen1unos_pg.sql

# 5. Ejecutar
python app.py
# → http://localhost:5000

    Respuesta JSON con Error Handling¶

    @app.route('/api/crear-vehiculo', methods=['POST'])
def crear_vehiculo():
    try:
        datos = request.get_json()

        # Validar
        if not datos.get('marca_id') or not datos.get('precio'):
            return jsonify({
                'success': False,
                'error': 'Campos requeridos'
            }), 400

        # Procesar
        vehiculo = Vehiculo(
            id_usuario=session['usuario_id'],
            id_marca=datos['marca_id'],
            precio=datos['precio']
        )
        db.session.add(vehiculo)
        db.session.commit()

        return jsonify({
            'success': True,
            'message': 'Vehículo creado',
            'vehiculo_id': vehiculo.id
        }), 201

    except Exception as e:
        db.session.rollback()
        logger.error(f"Error: {str(e)}")
        return jsonify({
            'success': False,
            'error': 'Error interno servidor'
        }), 500

    Proteger Ruta con Login¶

    from functools import wraps

def login_requerido(f):
    @wraps(f)
    def decorated_function(*args, **kwargs):
        if 'usuario_id' not in session:
            flash('Debes iniciar sesión', 'warning')
            return redirect(url_for('login'))
        return f(*args, **kwargs)
    return decorated_function

@users_bp.route('/dashboard')
@login_requerido
def dashboard():
    # Solo usuarios autenticados
    return render_template('dashboard.html')

    Procesar Archivos (Upload Imagen)¶

    from werkzeug.utils import secure_filename
import uuid

@vendedor_bp.route('/subir-imagen', methods=['POST'])
def subir_imagen():
    archivo = request.files.get('imagen')

    if not archivo or not allowed_file(archivo.filename):
        return jsonify({'error': 'Archivo no válido'}), 400

    try:
        # Generar nombre único
        extension = archivo.filename.rsplit('.', 1)[1].lower()
        nombre_unico = f"img_{uuid.uuid4().hex[:8]}.{extension}"

        # Guardar
        ruta = os.path.join('static/uploads/vehiculos', nombre_unico)
        archivo.save(ruta)

        return jsonify({
            'success': True,
            'url': f'/uploads/vehiculos/{nombre_unico}'
        })
    except Exception as e:
        logger.error(f"Upload error: {e}")
        return jsonify({'error': 'Error al subir'}), 500

    Validación de Datos¶

    # Crear validador personalizado
class ValidadorVehiculo:
    @staticmethod
    def validar(datos):
        errores = {}

        # Validar precio
        try:
            precio = float(datos.get('precio', 0))
            if precio <= 0:
                errores['precio'] = 'Debe ser positivo'
        except ValueError:
            errores['precio'] = 'Debe ser número'

        # Validar año
        anio = datos.get('anio')
        if anio and (anio < 1900 or anio > 2026):
            errores['anio'] = 'Año inválido'

        # Validar descripción
        desc = datos.get('descripcion', '')
        if len(desc) < 10:
            errores['descripcion'] = 'Mínimo 10 caracteres'

        return len(errores) == 0, errores

# Usar
valido, errores = ValidadorVehiculo.validar(datos)
if not valido:
    return jsonify({'success': False, 'errors': errores}), 400

    from flask_sqlalchemy import get_debug_queries

@app.after_request
def log_queries(response):
    for query in get_debug_queries():
        if query.duration >= 0.1:
            logger.warning(f"SLOW QUERY: {query.statement}\nTime: {query.duration}")
    return response

    Deploy a Heroku¶

    # Login
heroku login

# Crear app
heroku create drivemarket-prod

# Config vars
heroku config:set DATABASE_URL=postgresql://...
heroku config:set APP_SECRET_KEY=xxx

echo "web: gunicorn app:app" > Procfile

# Ver logs

    Deploy a AWS EC2¶

    # Conexión SSH
ssh -i clave.pem ubuntu@ip-instancia

# Instalar dependencias
sudo apt-get update
sudo apt-get install python3-pip postgresql nginx gunicorn

# Clonar código
git clone https://github.com/drivemarket/drivemarket.git
cd drivemarket

# Setup venv
python3 -m venv venv
source venv/bin/activate
pip install -r requirements.txt

# Configurar nginx
sudo cp nginx.conf /etc/nginx/sites-available/drivemarket
sudo systemctl restart nginx

# Ejecutar con gunicorn
gunicorn -w 4 -b 127.0.0.1:5000 app:app

# Con systemd (auto-restart)
sudo cp drivemarket.service /etc/systemd/system/
sudo systemctl start drivemarket
sudo systemctl enable drivemarket

    Info Sistema¶

    -- Tamaño BD
SELECT pg_size_pretty(pg_database_size('todoen1unos'));

-- Tablas grandes
SELECT relname, pg_size_pretty(pg_total_relation_size(relid))
FROM pg_stat_user_tables
ORDER BY pg_total_relation_size(relid) DESC;

-- Índices
SELECT * FROM pg_indexes WHERE tablename = 'vehiculos';

-- Conexiones activas
SELECT datname, count(*) FROM pg_stat_activity GROUP BY datname;

    Optimización¶

    -- Analizar tabla
ANALYZE vehiculos;

-- Reindex
REINDEX TABLE CONCURRENTLY vehiculos;

-- Vacuum (limpiar)

-- Ver query plan
EXPLAIN ANALYZE SELECT * FROM vehiculos WHERE precio > 1000000;
